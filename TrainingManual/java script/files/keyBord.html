События клавиатуры

        
            document.querySelector('.in').onkeypress = function (event) { // работает только для букв и некоторых вспомогательных клавиш
                console.log('keypress')
                console.log('charCode :' + event.charCode); устарело
                console.log('code :' + event.code);
                console.log('keyCode :' + event.keyCode); устарело
                console.log('key :' + event.key);
                console.log(event);

            }

            document.querySelector('.in').onkeydown = function (event) { // клавиша нажата но не отпущена работают буквы и другие кнопки
                console.log('keydown')
                console.log('charCode :' + event.charCode); устарело
                console.log('code :' + event.code);
                console.log('keyCode :' + event.keyCode); устарело
                console.log('key :' + event.key);
                console.log(event);

            }

             onkeypress  - устарело
             onkeyup  //событие отпускания клавиши
