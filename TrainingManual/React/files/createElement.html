import './App.css';

import React from 'react';

function App() {
    const cE = React.createElement;
    const title = React.createElement('h1', {}, 'Text Header 1'); //создание элемента и запись его в переменную h1-название элемента,{}-здесь передаются пропсы ,текст в нутри элемета необязательно
    const title2 = React.createElement('h2', {
    'color': 'red',  //передается как атребут но не работает
    'style': {        // стили сработывают можно использовать
        'color': 'red',
        'background': 'orange'
    },
    'id': 'hide', //можно задать id
    'className': 'italic bold' //можно задать классы
    }, 'Text Header 2');
    const s = cE('span', {}, "this is a span") //создаем элемент span и затем вставляем его в параграф
    const example = cE('p', {}, s);

    return (
    <>
        <React.Fragment>
        Hello
        </React.Fragment>
        {title}   {/*так вставляется  созданный элемент */}
        <div>
        {title2}   {/*так вставляется  созданный элемент */}
        </div>
        <div>{cE('input', {})}</div>   {/*создаем элемент сразу в диве */}
        <div>{example}</div>
    </>
    );
}

export default App;
