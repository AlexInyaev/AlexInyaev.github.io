State

State - это данные внутри компонента, изменение которых приводит к перерисовки компонента на странице. 

        
            import React from 'react';
            import { useState } from 'react';
            
            function Comp2() {
                const [count, setCount] = useState(0);    //для первого примера 
                const [comment, setComment] = useState([]); // для второго примера
                let myRef = React.createRef();
                // **************************************************** Первый пример
                const handler = () => {
                    let currentCount = count;
                    currentCount++;
                    setCount(currentCount);          //ИЗМЕНЯЕС СТЭЙТ И ПЕРЕЗАПИСЫВАЯ КОМПОНЕНТ
                }
            
                // *********************************** Второй пример
            
                const addComment = () => {
                    let commentValue = myRef.current.value;        // получаем commentValue с инпут через реф 
                    let comments = [...comment, commentValue];     // кладем в comments массив comment плюс значение из commentValue полученое из инпут
                    setComment(comments);                      // кладем в comment comments ТЕМ САМЫМ ИЗМЕНЯЕС СТЭЙТ И ПЕРЕЗАПИСЫВАЯ КОМПОНЕНТ
                    myRef.current.value = '';
                }
                return (
                    <>
                        {/* Верстка для первого примера */}
                        <h1>State</h1>
                        <div><button onClick={handler}>This button change state</button></div>
                        <div>{count % 2 === 0 ? 'even' : 'odd'}</div>
                        <div>{count}</div>
            
                        {/* Верстка для второго примера */}
                        <div>
                            <textarea ref={myRef}></textarea>
                            <button onClick={addComment}>Add comment</button>
                        </div>
                        <div>
                            <ul>
                                {comment.map((item, index) => <li key={index.toString()}>{item}</li>)}
                            </ul>
                        </div>
                    </>
                );
            }
            
            export default Comp2
