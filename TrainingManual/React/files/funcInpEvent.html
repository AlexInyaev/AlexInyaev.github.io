            import './App.css'
            import React from 'react';             подключение реф
            
            
            function App() {
            
                let textInput = React.createRef(); получаем реф в переменную
                let textOut = React.createRef();   получаем реф в переменную для вывода
            
                function showInput(event) {
                    console.log('input');
                    // console.log(this.value);   не работает
                    console.log(event.target.value);         вывод в консоль через  event
                    
                    console.log(textInput.current.value);    вывод в консоль через  реф
                    textOut.current.innerHTML = textInput.current.value;   Вывод через реф  в параграф где ref = { textOut }
                }
            
            
                return (
                    <>
                        <section>
                            <h2>Input get value by ref</h2>
                            <input type="text" onInput={showInputRef} ref={textInput} />      ref={textInput}-прописываем для дальнейшего получения значения получаем инпут в переменную
                            <p ref={textOut}></p>       Вывод в параграф через реф    ref={textOut} получаем параграф в textOut
                           
                        </section>
                    </>
                );
            }
            
            export default App;

Вывод через стэйт

        
            import './App.css'
            import { useState } from 'react';
            import React from 'react'; 
            
            function App() {
            
                const [output, setOutput] = useState('hello');          output - константа, setOutput - метод который записывает в output значение,
                                                                         useState('hello') - вскобках значение по умолчанию можно оставить пустые скобки
            
                function showInput() {
                    setOutput(textInput.current.value);                 В setOutput() мы кладем то что хотим вывести => output => h3
                }
            
            
                return (
                    <>
                        <section>
                            <h2>Input get value by ref</h2>
                            <input type="text" onInput={showInput} ref={textInput} />
            
                            <h3>{output}</h3>                          Вывод в заголовок через стэйт   output- значение стэйта
                        </section>
                    </>
                );
            }
            
            export default App;
