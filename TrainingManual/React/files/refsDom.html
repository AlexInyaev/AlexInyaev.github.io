Refs and Dom

        
            import { useRef } from "react";
            import Test3 from "./Tast3";

            function Test() {
                let refTest = useRef();  - создаем реф

                function tst3() {
                    refTest.current.drive(); - вызываем метод у компонента через реф ВОЗМОЖНО ТОЛЬКО У ООП компонентах
                }

                return (
                    <div>
                        <Test3 ref={refTest} />      - через реф получаем ссылку на компонент
                        <button onClick={tst3}>ts3</button>
                    </div>
                )
            }
            
            export default Test;
