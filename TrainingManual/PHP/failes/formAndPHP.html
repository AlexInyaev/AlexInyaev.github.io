Передача данных из формы в php

        
        <form action="main.php" method="get">   // action="main.php" - файл обработчик
            <input type="text" name="a1">       // name="a1" - имя инпут обязательно в GET будет ключом
            <input type="text" name="a2">
            <input type="submit" value="push">  // кнопка отправки формы
        </form>

        после нажатия кнопки все данные передаются в супер глобальный массив  GET

PHP

        
            <?php
            var_dump($_GET);  // информация о переменной
            print_r($_GET);   // вывод объктов и массивов
            
            $a1 = $_GET[a1];  // присвоение значения с ключем а1 из массива GET в переменную
            $a2 = $_GET[a2];
            
            if($a1 == $a2) {  //  программа вывода наибольшего числа
                echo "$a1 = $a2";
            } else if ($a1 > $a2) {
                echo "$a1 > $a2";
            } else if($a1 < $a2) {
                echo "$a1 < $a2";
            }

Проверка вводимых данных

        
            function checkVar($a){
                if (isset($a) AND trim($a) !=''AND is_numeric($a)){
                    return trim($a);
                }
                else {
                    exit('Problem!!!');
                }
            }

            /*
            isset()  -  Возвращает TRUE, если var определена и ее значение отлично от NULL, и FALSE в противном случае.
            trim()   -  Удаляет пробелы (или другие символы) из начала и конца строки
            is_numeric — Проверяет, является ли переменная числом или строкой, содержащей число
            */

            $n1 = checkVar($_GET['n1']);
            $n2 = checkVar($_GET['n2']);
