            <?php

            $dir = 'img';    // получение папки
            
            if (is_dir($dir)){  // проверка на наличие папки
                $openDir = opendir($dir); // открываем директорию
            while($file = (readdir($openDir))){  // в $file присваеваем последующие значение из диектории и сравниваем с фолс
                                                         // если файлов нет то будет фолс
                if($file != "." && $file != ".."){     // "." проверка на ссылку на папку и на ссылку на ".." родителя
                    echo '<img src="'.$dir.'/'.$file.'" alt="imge">';  // $dir - название папки  $file - название файла
                }
               
            }
            
                closedir($openDir); // закрываем директорию
            }
            
            
            ?>
