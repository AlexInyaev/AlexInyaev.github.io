Работа с файлом

        
            <?php 
                                                                        //ЧТЕНИЕ
            $filename = 'file.txt'; 
                                                                        // СПОСОБ1
            
            if (file_exists($filename)){   //проверка на существование файла
               $handle = fopen($filename, "r");  //открытие файла для чтения
                $contents = fread($handle, filesize($filename)); // чтение файла в переменную filesize($filename)- размер файла может быть просто цифра размера в символах
                echo $contents;                                 // вывод файла
               fclose($handle);      //закрытие файла
            
            }
            
                                                                        //СПОСОБ2
            $second = file_get_contents($filename);
            echo'<hr>';
            echo $second;
            
            
                                                                        //ЗАПИСЬ


                                                                        // СПОСОБ1
            
            if (file_exists($filename)){ //проверка на существование файла
                
                $handle = fopen($filename, "a"); //открытие для записи
                fwrite($handle, 'bla bla');       // запись
            
            
                fclose($handle);
            }
            // r   -   открытие файла на чтение
            // r+  -   ставит курсор в начало и затирает текст в начале
            // w   -   заминяет текст
            // a   -   дописывает текст в конец
                                                                        // СПОСОБ2
            file_put_contents($filename, 'second write',FILE_APPEND);
            ?>
