функция проверки чисел

        
    function checkVar($v){
        if(trim($v)!=''AND isset($v)AND is_numeric($v)){
            return trim($v);
        }
        else{
            exit ('problem!!!');
        }
    }

            
trim();     обрезает пробелы
isset() ;  проверка существования переменной
is_int(); проверка на целое число
is_numeric(); проверка строки на содержание в ней целого числа
exit();    выводит сообщение из скобок и останавливает скрипт
