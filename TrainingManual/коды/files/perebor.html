            <script> 
            let active = 0;
            const  image = document.querySelectorAll('img'); // получаем коллекцию картин
        
            document.querySelector('.but').onclick = () => {
            image[active].classList.remove('act');    //удоляем класс

            if(active + 1 == image.length){                  // чтобы н уходило за границы
                active = 0;
            }else{
            active++;
            }

            image[active].classList.add('act'); //добовляем класс
            }
