          CREATE DATABASE shop;  - создание базы данных с названием shop
          DROP DATABASE shop;    - удаление базы данных с названием shop 
          
          --Создание таблицы --

          CREATE TABLE people(
              id INT NOT NULL AUTO_INCREMENT,
              name VARCHAR(30),
              bio TEXT,
              birth DATE,
              PRIMARY KEY(ID)
          );

          people() - название таблицы в скобках пишем сначала поле затем атрибуты этого поля. Названия полей -- id name bio birth --
          INT - тип данных
          NOT NULL - не может быть пустым
          AUTO_INCREMENT - автозаполнение 
          VARCHAR(30) - тип данных в скобках прописывается длинна в символах
          TEXT - тип данных
          DATE - тип данных
          PRIMARY KEY(ID) - поле будет уникальным

          -- Удаление таблицы --
          DROP TABLE shop
          shop - название таблицы

          -- Добавление полей в таблицу --

          ALTER TABLE people ADD pass VARCHAR(32);

          people - название таблицы
          pass - название поля
          VARCHAR(32); - тип данных и 32 -мы используем потому-что при кодировке пароля длинна хеша 32

          -- Удаление поля таблицы --
          ALTER TABLE people DROP COLUMN pass;

          people - название таблицы
          pass - название удаляемого поля
