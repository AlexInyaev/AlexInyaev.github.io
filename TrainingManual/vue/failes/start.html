события классы переменные атрибуты

        
            <div id="app">
            <p>{{title}}</p> //вывод переменной
            <button v-on:click="title='v-on'">v-on</button> //событие
            <button @click="title='@'">@</button> //событие короткая запись
            <button @click="useFunction">function</button> событие и метод из массива vue
    
            <p v-bind:class="styleCss">{{styleCss}}</p> бандим класс
            <p :class="styleCss">{{styleCss}}</p> бандим класс короткая запись
            <input type="text" v-on:input="styleCss = $event.target.value"> записываем в переменную значение из объекта событий и выводим ее в параграфах

            <img v-bind:src="address" :alt="alt"> добавление атрибутов address и alt это свойства из массива vue если поставить строку то будет ошибка

        </div>


        код js

        let app = new Vue({
            el: '#app',
            data: {
                title: "hello world",
                styleCss: "",
                address: 'https://cdn-icons-png.flaticon.com/512/254/254232.png',
                alt: "просто картинка"
            },
            methods: {
                useFunction() {
                    this.title = 'use function'
                }
            },
        
        })
