            <div id="app">

            <input type="text" v-on:input="increment($event.target.value)">
            <p>{{value}}</p>
            <p>{{dubleValue}}</p>
            <p>valueFromComputed: {{valueFromComputed}}</p>
    
    
        </div>


        код js

        let app = new Vue({
            el: '#app',
            data: {
                value: 1,
                dubleValue: 1
            },
            methods: {
                increment(value) {
                    this.value = value,
                        this.dubleValue = value * 2
                    // this.dubleValue =  this.value * 2 // второй вариант записи устанавливает связь между переменными
                }
            },
            computed: {                 //вычисляемые свойства
                valueFromComputed() {
                    return this.value * 3;
                }
            }
        
        })
