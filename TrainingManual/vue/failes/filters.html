Фильтры

        
            <div id="app">

            <p>При помощи computed:{{showMassage}}</p>
            <p>При помощи локального фильтра:{{showMassage | lowercase}}</p> <!-- пишется через | -->
            <p>При помощи глобального фильтра:{{showMassage | capitalize }}</p>  

        </div>

        js code 

        Vue.filter('capitalize', function (value) { //глобальный фильтр можно применять и в других компонентах
            return value.toUpperCase();
        });
        
        let app = new Vue({
            el: '#app',
            data: {
                massage: 'hello'
            },
        
            methods: {
        
            },
            computed: {
                showMassage() {
                    return this.massage.toUpperCase();
                }
            },
            filters: {
                lowercase(value) {              //локальный фильтр может применятся к любой переменной только в этом компоненте
                    return value.toUpperCase();
                }
            }
        
        })
