        js код

        // Глобальный компонент его можно применять к любым Vue объектам 
Vue.component('component-name', {
    data: function () {
        return {
            ggg: 'эта функция возвращает объект данных'
        }
    },
    template: '
{{ggg}}
'
});
new Vue({ el: '#app' })

// локальный компонент

new Vue({
    el: '#app2',
    components: {
        'component-name2': {
            data: function () {
                return {
                    ggg: 'эта функция возвращает объект данных2'
                }
            },
            template: '
{{ggg}}
'
        }
    }
})
