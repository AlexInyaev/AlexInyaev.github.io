Основы ООП создание классов и объектов

        
            <?php
            class User{                //создание класса название класса пишется с большой буквы это правило хорошего тона
             public $name = 'None';
             private $surname;
             private $email;
             private $login;
             private $pass;
            }
            //модификаторы доступа 
            //1) public - данное поле видно в пределах класса и не в класса
            //2) protected - видно в данном классе и классах наследниках
            //3) private - видно только внутри  класса
         
            $admin = new User(); //создал объект на основе класса User
            $admin ->$name = 'John'; //присвоил строку в поле name
            echo $admin->$name;
         
            $admin2 = new User(); //создал второй объект на основе класса User
            $admin ->$name = 'Jack'; //присвоил строку в поле name
            echo $admin->$name;
         
            /*
            1) Наследование - можем унаследовать и первого класса все его свойства во второй
            2) Полиморфизм - возможность переопределять различные свойства в классах наследниках
            3) Инкапсуляция - возможность открывать и закрывать поля в классах
            
            
            */
             ?>
