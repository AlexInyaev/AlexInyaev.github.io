    display: block;             - блок
    display: inline;            - строка
    display: inline-block;      - блок - строка
    display: table;             - блочная таблица
    display: table-cell;        - ячейка таблици
    display: none;              - нет
    display: flex;              - flexbox элемент
    display: grid;              - grid элемент
